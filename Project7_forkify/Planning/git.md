## Run git to version control
# init git to project
git init

# add change of file
git add -A

# check git status
git status

# commit message(string) to change file
git commit -m 'commit message'

# Return previously version
git reset --hard [HEAD]

# reading all commit
git log

# reset previouly version
git reset --hard [commit id]

# check branch
git branch

# create new branch
git branch [branch name]

# switch branch
git checkout [branch name]

# merge branch to current branch
git merge [branch name]

# push change
git push [brand naem like: origin] [branch name]

# add new branch to remote
git push [brand naem like: origin] [new branch name]

# push to github repo flow
git add -A
git commit -m <commit-message>
git push origin <branch-name>
